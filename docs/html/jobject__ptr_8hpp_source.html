125  template<typename Result, typename Func, typename... Args>
126  Result call_method(const std::string& methodname, const std::string& signature,
127  Func func, Args... args) {
128  auto env = thread_local_jni_env_ptr();
129  jmethodID method = env->GetMethodID(cls.get(), methodname.c_str(), signature.c_str());
130  if (nullptr == method) {
131  throw jni_exception(TRACEMSG("Cannot find method, name: [" + methodname + "]," +
132  " signature: [" + signature + "], class: [" + cls.name() + "]"));
133  }
134  Result res = (env.get()->*func)(obj.get(), method, args...);
135  if (env->ExceptionCheck()) {
136  env->ExceptionClear();
137  throw jni_exception(TRACEMSG("Exception raised when calling method,"
138  " name: [" + methodname + "], signature: [" + signature + "], class: [" + cls.name() + "]"));
139  }
140  return res;
141  }
