63  cls([this] {
64  auto env = thread_local_jni_env_ptr();
65  jclass clazz_local = env->FindClass(clsname.c_str());
66  if (nullptr == clazz_local) {
67  throw jni_exception(TRACEMSG("Cannot load class, name: [" + clsname + "]"));
68  }
69  auto clazz = std::unique_ptr<_jclass, std::function<void(jclass)>>(clazz_local, [&env] (jclass cl) {
70  env->DeleteLocalRef(cl);
71  });
72  jclass clazz_global = static_cast<jclass> (env->NewGlobalRef(clazz.get()));

74  throw jni_exception(TRACEMSG("Cannot create global ref for class, name: [" + clsname + "]"));

141  Result call_static_method(const std::string& methodname, const std::string& signature,
142  Func func, Args... args) {
143  auto env = thread_local_jni_env_ptr();
144  jmethodID method = env->GetStaticMethodID(cls.get(), methodname.c_str(), signature.c_str());
145  if (nullptr == method) {
146  throw jni_exception(TRACEMSG("Cannot find method, name: [" + methodname + "]," +
147  " signature: [" + signature + "], class: [" + clsname + "]"));
148  }
149  Result res = (env.get()->*func)(cls.get(), method, args...);
150  if (env->ExceptionCheck()) {
151  env->ExceptionClear();
152  throw jni_exception(TRACEMSG("Exception raised when calling method,"
153  " name: [" + methodname + "], signature: [" + signature + "], class: [" + clsname + "]"));
154  }
